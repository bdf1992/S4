# files that name a key not in this taxonomy — typo-detection at the
# field level, identical in shape to leash_for_hooks's hook-event
# membership check.
#
# Format: one dotted field-path per line, lowercase. Lines starting with
# "#" are comments; blank lines are skipped.
#
# Sources, commit-pinned at proposal write-time:
#   - hawkymisc/cc-symphony SPEC_GITHUB.md + WORKFLOW.md — verified
#     2026-05-08

# Updating this file is itself a 0.4 grading event. The line-set is the
# 0.1 dataset; verify.py confirms the data points the collector emits
# match the file as committed.

# Body template variable taxonomy (Liquid-compatible, strict).
#
# The WORKFLOW.md body (below the front-matter fence) is a Liquid template.
# Only the variables listed below are injected by cc-symphony. References
# to any other variable name will render as empty string without error.
#
# Issue-level variables:
#   {{ issue.id }}          — internal UUID
#   {{ issue.identifier }}  — human-readable ID (e.g. "4")
#   {{ issue.title }}       — issue title string
#   {{ issue.description }} — issue body markdown
#   {{ issue.priority }}    — priority label string (e.g. "medium")
#   {{ issue.state }}       — current state name (e.g. "In Progress")
#   {{ issue.branch_name }} — git branch cc-symphony provisioned
#   {{ issue.url }}         — URL to the issue on the tracker
#   {{ issue.labels }}      — list of label strings
#   {{ issue.blocked_by }}  — list of blocking issue identifiers
#   {{ issue.created_at }}  — ISO 8601 timestamp
#   {{ issue.updated_at }}  — ISO 8601 timestamp
#
# Run-level variables:
#   {{ attempt }}           — integer retry count (1 on first dispatch)
#
# Tracker-level convenience:
#   {{ repo }}              — the tracker.repo value ("owner/repo")
# ---------------------------------------------------------------------------
